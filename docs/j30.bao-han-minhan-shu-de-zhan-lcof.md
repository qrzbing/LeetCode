```cpp
class MinStack
{
public:
    /** initialize your data structure here. */
    stack<int> st;
    stack<int> min_st;
    MinStack()
    {
        while (!st.empty())
            st.pop();
        while (!min_st.empty())
            min_st.pop();
    }

    void push(int x)
    {
        st.push(x);
        if (min_st.empty())
        {
            min_st.push(x);
        }
        else
        {
            if (x <= min_st.top())
            {
                min_st.push(x);
            }
        }
    }

    void pop()
    {
        if (st.top() == min_st.top())
        {
            min_st.pop();
        }
        st.pop();
    }

    int top()
    {
        return st.top();
    }

    int min()
    {
        return min_st.top();
    }
};
```
